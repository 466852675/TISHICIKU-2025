class InformationSearcher {
    async searchAcrossPlatforms(query) {
        const platforms = [
            'bing.com', 'baidu.com', 'weibo.com',
            'toutiao.com', 'bilibili.com', 'juejin.cn'
        ];

        const results = [];
        for (const platform of platforms) {
            const data = await this.searchPlatform(query, platform);
            results.push(...data);
        }

        return this.filterAndSortResults(results);
    }

    filterAndSortResults(results) {
        // 信息质量筛选和排序逻辑
        return results
            .filter(item => this.isValidSource(item))
            .sort((a, b) => this.calculateScore(b) - this.calculateScore(a));
    }
}

### 信息搜索整合模板

```javascript
// 信息搜索与整合系统
class WebPageGenerator {
    constructor() {
        this.searchPlatforms = [
            'bing.com', 'baidu.com', 'weibo.com',
            'toutiao.com', 'bilibili.com', 'juejin.cn',
            'tencent.com', 'thepaper.cn'
        ];
    }

    async generateWebPage(userQuery) {
        try {
            // 1. 信息搜索阶段
            const searchResults = await this.searchAcrossPlatforms(userQuery);

            // 2. 信息整合阶段
            const integratedContent = await this.integrateInformation(searchResults);

            // 3. 网页生成阶段
            const htmlContent = await this.generateHTML(integratedContent);

            return {
                success: true,
                htmlContent: htmlContent,
                searchSummary: this.generateSearchSummary(searchResults),
                qualityScore: this.calculateQualityScore(htmlContent)
            };

        } catch (error) {
            return {
                success: false,
                error: error.message
            };
        }
    }

    async searchAcrossPlatforms(query) {
        const results = [];

        for (const platform of this.searchPlatforms) {
            try {
                const platformResults = await this.searchPlatform(query, platform);
                results.push(...platformResults);
            } catch (error) {
                console.warn(`搜索平台 ${platform} 失败:`, error.message);
            }
        }

        return this.filterAndSortResults(results);
    }

    filterAndSortResults(results) {
        return results
            .filter(item => this.isValidSource(item))
            .filter(item => this.isRecentContent(item))
            .sort((a, b) => this.calculateRelevanceScore(b) - this.calculateRelevanceScore(a))
            .slice(0, 50); // 限制结果数量
    }

    isValidSource(item) {
        // 验证信息源的可靠性
        const trustedDomains = ['gov.cn', 'edu.cn', 'com', 'org'];
        return trustedDomains.some(domain => item.url.includes(domain));
    }

    isRecentContent(item) {
        // 检查内容的时效性（最近30天）
        const thirtyDaysAgo = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000);
        return new Date(item.publishDate) > thirtyDaysAgo;
    }
}
```
